layout FrameLayout { 
	slot CameraData;
	slot MaterialData;
}



struct Camera
{
	matrix view;
	matrix proj;
	matrix viewProj;
	matrix invView;
	matrix invProj;
	matrix invViewProj;
	float3 position;
	float unused;
	float3 direction;
	float unused2;
	float2 jitter; float2 unused3;
}


[Bind = FrameLayout::CameraData]
struct FrameInfo
{
    Camera camera;
    Camera prevCamera;
}


[Bind = FrameLayout::MaterialData]
struct Material
{
    uint textureOffset;
    uint nodeOffset;
}
